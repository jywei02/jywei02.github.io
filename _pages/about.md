            const v = parseInt((el.textContent || '0').replace(/,/g,''), 10);

          }
        }
      });
      obs.observe(el, { characterData: true, childList: true, subtree: true });

          clearInterval(t);
          flipOnce();
        }
        if (++tries > 50) clearInterval(t);
      }, 100);
